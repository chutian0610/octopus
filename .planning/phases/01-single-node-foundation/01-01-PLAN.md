```rust
use apache_datafusion::prelude::*;
use apache_datafusion::execution::runtime_env::{RuntimeConfig, RuntimeEnv};
use std::sync::Arc;
use async_trait::async_trait;
use crate::error::{Result, OctopusError};
use tracing::info;

/// Session wrapper for DataFusion execution with proper runtime isolation.
/// 
/// Per D-02: CPU-intensive work runs on separate thread pool to avoid
/// blocking async I/O tasks. Uses DataFusion's multi-threaded runtime.
pub struct ExecutorSession {
    context: SessionContext,
}

impl ExecutorSession {
    /// Create new session with DataFusion defaults.
    /// Target partitions = number of available CPUs.
    pub fn new() -> Result<Self> {
        let config = SessionConfig::new()
            .with_target_partitions(num_cpus::get())
            .with_information_schema(true);
        
        let runtime = Arc::new(
            RuntimeEnv::new(RuntimeConfig::default())
                .map_err(|e| OctopusError::ExecutionError(e.to_string()))?
        );
        
        let state = SessionState::new(config, runtime, ExecutionProperties::new());
        let context = SessionContext::with_state(state);
        
        info!("ExecutorSession created with {} target partitions", num_cpus::get());
        
        Ok(Self { context })
    }
    
    /// Get reference to underlying DataFusion session context.
    pub fn context(&self) -> &SessionContext {
        &self.context
    }
}

impl Default for ExecutorSession {
    fn default() -> Self {
        Self::new().expect("Failed to create default ExecutorSession")
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    
    #[tokio::test]
    async fn test_session_creation() {
        let session = ExecutorSession::new();
        assert!(session.is_ok());
    }
}
```

Note: For Phase 1, we use DataFusion's built-in runtime. The separate CPU/IO runtime separation (D-02) will be implemented when integrating with Arrow Flight in Phase 3.
