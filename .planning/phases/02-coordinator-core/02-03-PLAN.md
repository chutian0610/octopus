```rust
async fn run_batch(cli: Cli) -> anyhow::Result<()> {
    let client = CoordinatorClient::connect(&cli.coordinator).await?;

    let sql_lines: Box<dyn Iterator<Item = String>> = if let Some(file_path) = &cli.file {
        let content = std::fs::read_to_string(file_path)
            .map_err(|e| anyhow::anyhow!("Failed to read file {}: {}", file_path, e))?;
        let lines = content.lines()
            .map(|s| s.to_string())
            .collect::<Vec<_>>();
        Box::new(lines.into_iter())
    } else {
        // Read from stdin
        let mut lines = Vec::new();
        for line in std::io::stdin().lines() {
            lines.push(line?);
        }
        Box::new(lines.into_iter())
    };

    // Execute each statement (separated by semicolons)
    let mut current_stmt = String::new();
    for line in sql_lines {
        let trimmed = line.trim();
        if trimmed.is_empty() || trimmed.starts_with("--") {
            continue;  // Skip empty lines and comments
        }
        
        current_stmt.push_str(line.as_str());
        current_stmt.push(' ');
        
        if trimmed.ends_with(';') {
            // Execute the statement
            let stmt = current_stmt.trim_end_matches(';');
            if !stmt.is_empty() {
                if stmt.to_lowercase().starts_with("explain") {
                    let sql = &stmt[7..].trim();
                    let plan = client.explain_query(sql).await?;
                    println!("{}", plan);
                } else {
                    let query_id = client.submit_query(stmt).await?;
                    println!("Query {}: {}", query_id, stmt);
                }
            }
            current_stmt.clear();
        }
    }

    Ok(())
}
```
