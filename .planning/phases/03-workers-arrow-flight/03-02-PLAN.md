//! Arrow Flight server for worker data exchange.
//!
//! Workers expose Flight servers; consumers pull data on demand (worker-pull model).
//! This enables Exchange operators to stream data between workers.

use std::sync::Arc;
use tonic::{Request, Response, Status, Streaming};
use arrow_flight::{
    flight_service_server::{FlightService, FlightServiceServer},
    FlightData, FlightDescriptor, FlightEndpoint, Ticket,
    SchemaResult, PutResult, Fingerprint,
};
use arrow::schema::Schema;
use crate::runtime::IoRuntime;
use crate::task_processor::TaskProcessor;
use crate::flight_handler::FlightHandler;
use tracing::info;

/// Arrow Flight server for worker data exchange.
pub struct FlightServer {
    worker_id: String,
    port: u16,
    io_runtime: Arc<IoRuntime>,
    handler: Arc<FlightHandler>,
}

impl FlightServer {
    pub fn new(
        worker_id: String,
        port: u16,
        io_runtime: Arc<IoRuntime>,
        handler: Arc<FlightHandler>,
    ) -> Self {
        Self {
            worker_id,
            port,
            io_runtime,
            handler,
        }
    }

    /// Start the Flight server and return the bind address.
    pub async fn start(&self) -> Result<String, Box<dyn std::error::Error + Send + Sync>> {
        let addr = format!("0.0.0.0:{}", self.port).parse()?;
        
        info!("Starting Arrow Flight server on {}", addr);

        let service = FlightService::new(self.clone());
        let server = FlightServiceServer::builder(service)
            .addr addr)
            .build()?;

        info!("Arrow Flight server started on {}", addr);
        
        Ok(addr.to_string())
    }

    /// Create a FlightEndpoint for accessing exchange data.
    pub fn create_endpoint(&self, ticket: Ticket) -> FlightEndpoint {
        FlightEndpoint {
            ticket: Some(ticket),
            location: vec![],  // Will be populated with actual address
        }
    }
}

impl Clone for FlightServer {
    fn clone(&self) -> Self {
        Self {
            worker_id: self.worker_id.clone(),
            port: self.port,
            io_runtime: self.io_runtime.clone(),
            handler: self.handler.clone(),
        }
    }
}

impl FlightService for FlightServer {
    /// Get a Flight schema (metadata only, no data).
    async fn get_flight_schema(
        &self,
        request: Request<FlightDescriptor>,
    ) -> Result<Response<SchemaResult>, Status> {
        let descriptor = request.into_inner();
        info!("Schema request for {:?}", descriptor);
        
        let schema = self.handler.get_schema(&descriptor)?;
        let schema_result = SchemaResult { schema: schema.to_bytes().map_err(|e| {
            Status::internal(format!("Failed to encode schema: {}", e))
        })? };

        Ok(Response::new(schema_result))
    }

    /// DoGet: Fetch data for a ticket (used by consumers to pull exchange data).
    async fn do_get(
        &self,
        request: Request<Ticket>,
    ) -> Result<Response<Streaming<FlightData>>, Status> {
        let ticket = request.into_inner();
        info!("DoGet request for ticket: {:?}", ticket);
        
        // Spawn the data stream generation on the IO runtime
        let handler = self.handler.clone();
        let ticket_bytes = ticket.ticket.clone();

        let stream = futures::stream::unfold(0, move |batch_idx| {
            let handler = handler.clone();
            let ticket_bytes = ticket_bytes.clone();
            async move {
                // Generate flight data for the requested batch
                // In real implementation, this would stream from task execution
                match handler.get_batch(&ticket_bytes, batch_idx) {
                    Some(data) => Some((Ok(data), batch_idx + 1)),
                    None => None,  // End of stream
                }
            }
        });

        Ok(Response::new(stream))
    }

    /// DoPut: Upload data (used for shuffle data, etc).
    async fn do_put(
        &self,
        request: Request<Streaming<FlightData>>,
    ) -> Result<Response<PutResult>, Status> {
        let mut stream = request.into_inner();
        
        while let Some(data) = stream.next().await {
            match data {
                Ok(d) => {
                    self.handler.put_batch(d)?;
                }
                Err(e) => {
                    return Err(Status::internal(format!("DoPut error: {}", e)));
                }
            }
        }

        Ok(Response::new(PutResult { app_metadata: vec![] }))
    }

    /// ListFlights: Discover available flights (queries/stages).
    async fn list_flights(
        &self,
        request: Request<FlightDescriptor>,
    ) -> Result<Response<Streaming<FlightDescriptor>>, Status> {
        let descriptors = self.handler.list_flights()?;
        let stream = futures::stream::iter(descriptors.into_iter().map(Ok));
        Ok(Response::new(stream))
    }

//! Worker service with integrated Arrow Flight data plane.

use std::sync::Arc;
use crate::runtime::{WorkerRuntime, IoRuntime};
use crate::task_processor::TaskProcessor;
use crate::flight_server::FlightServer;
use crate::flight_handler::FlightHandler;
use octopus_common::Result;

pub struct WorkerService {
    worker_id: String,
    runtime: Arc<WorkerRuntime>,
    processor: Arc<TaskProcessor>,
    flight_server: Arc<FlightServer>,
    coordinator_url: String,
}

impl WorkerService {
    pub fn new(coordinator_url: String, flight_port: u16) -> Result<Self> {
        let runtime = Arc::new(WorkerRuntime::new()?);
        let processor = Arc::new(TaskProcessor::new(runtime.cpu.clone())?);
        let handler = Arc::new(FlightHandler::new(processor.clone()));
        let flight_server = Arc::new(FlightServer::new(
            Uuid::new_v4().to_string(),
            flight_port,
            runtime.io.clone(),
            handler,
        ));

        let worker_id = flight_server.worker_id.clone();

        tracing::info!("WorkerService created with ID: {}", worker_id);

        Ok(Self {
            worker_id,
            runtime,
            processor,
            flight_server,
            coordinator_url,
        })
    }

    pub fn worker_id(&self) -> &str {
        &self.worker_id
    }

    /// Start the worker service including Flight server.
    pub async fn run(&self) -> Result<()> {
        // Start Arrow Flight server on IO runtime
        let flight_addr = self.flight_server.start().await?;
        tracing::info!("Arrow Flight server running on {}", flight_addr);

        // Register with coordinator
        self.register_with_coordinator().await?;

        // Run task receiver loop
        self.task_receiver_loop().await;

        Ok(())
    }

    async fn register_with_coordinator(&self) -> Result<()> {
        tracing::info!("Registering worker {} with coordinator at {}",
                      self.worker_id, self.coordinator_url);
        Ok(())
    }

    async fn task_receiver_loop(&self) {
        tracing::info!("Task receiver loop started for worker {}", self.worker_id);
        
        loop {
            tokio::time::sleep(tokio::time::Duration::from_secs(5)).await;
        }
    }
}
